  _collect_agent_progress(session, date)  -> dict (Section 2: agent statuses, pending approvals)
  _collect_market_data(session, date)     -> dict (Section 3: market size, competitor alerts, inventory)
  generate_feishu_card(report_data)       -> dict (Feishu card JSON with config/header/elements)
  generate_daily_report(dry_run=True)     -> dict (full report, writes audit log, saves to DB)
  _save_to_db(session, report_data)       -> None (upsert agent_runs + daily_reports tables)
  _mark_report_sent(session, run_id)      -> None (update sent_at after Feishu delivery)

  DailyReportAgent(BaseAgent)
    AGENT_TYPE = "daily_report_agent"
    run()  -> executes full flow: collect data -> generate card -> send Feishu -> mark sent
